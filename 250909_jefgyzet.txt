namespace konyvek
{
    public class Kiadasok
    {
        public int year;
        public int part;
        public string type;
        public string desc;
        public int count;

        public Kiadasok(string line)
        {
            string[] strings = line.Split(';');
            this.year = int.Parse(strings[0]);
            this.part = int.Parse(strings[1]);
            this.type = strings[2];
            this.desc = strings[3];
            this.count = int.Parse(strings[4]);
        }
    }
    internal class Program
    {
        static void Main(string[] args)
        {
            Console.WriteLine("1. Feladat");
            List<Kiadasok> kiadasok = new List<Kiadasok>();

            StreamReader sr = new StreamReader("kiadas.txt", Encoding.UTF8);

            while (!sr.EndOfStream)
            {
                Kiadasok tmp = new Kiadasok(sr.ReadLine()); { kiadasok.Add(tmp); }
            }

            sr.Close();

            if (kiadasok.Count > 0) { Console.WriteLine("\t" + kiadasok.Count + " adat sikeresen beolvasva!"); } else { Console.WriteLine("\tSikertelen adatbeolvasás)"); }

        }
    }
}
